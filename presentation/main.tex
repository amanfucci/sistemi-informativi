\begin{frame}[t]{Cybersecurity Policies}
    \begin{multicols}{2}

            \begin{itemize}
                \itemj{} anno
                \itemj{} nazione
                \itemj{} policy sicurezza
                \itemj{} grandezza dell'azienda
            \end{itemize}
